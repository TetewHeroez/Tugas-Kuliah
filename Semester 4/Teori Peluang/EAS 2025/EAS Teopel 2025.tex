\begin{center}
  \textbf{SOLUSI}
\end{center}
\begin{enumerate}
  \item Karena $X_1$ dan $X_2$ adalah sampel acak berukuran $n=2$, maka $X_1$ dan $X_2$ saling bebas dan berdistribusi sama (i.i.d). Oleh karena itu, pdf bersama adalah $f(x_1, x_2) = f(x_1)f(x_2)$.
        \begin{enumerate}[label=\alph*)]
          \item Tabel pdf bersama $f(x_1, x_2)$:
                \begin{center}
                  \begin{tabular}{|c|c|c|c|}
                    \hline
                    $x_1 \backslash x_2$ & \textbf{1}              & \textbf{2}              & \textbf{3}              \\ \hline
                    \textbf{1}           & $0.2 \times 0.2 = 0.04$ & $0.2 \times 0.3 = 0.06$ & $0.2 \times 0.5 = 0.10$ \\ \hline
                    \textbf{2}           & $0.3 \times 0.2 = 0.06$ & $0.3 \times 0.3 = 0.09$ & $0.3 \times 0.5 = 0.15$ \\ \hline
                    \textbf{3}           & $0.5 \times 0.2 = 0.10$ & $0.5 \times 0.3 = 0.15$ & $0.5 \times 0.5 = 0.25$ \\ \hline
                  \end{tabular}
                \end{center}
          \item $P[X_1 + X_2 \le 3]$ adalah peluang jumlah $X_1$ dan $X_2$ kurang dari atau sama dengan 3. Kemungkinan yang memenuhi adalah $(1,1), (1,2),$ dan $(2,1)$.
                \begin{align*}
                  P[X_1 + X_2 \le 3] & = f(1,1) + f(1,2) + f(2,1) \\
                                     & = 0.04 + 0.06 + 0.06       \\
                                     & = 0.16
                \end{align*}
          \item Dapatkan $E[2X_1 + 3X_2 + 1]$ dan $Var[2X_1 + 3X_2 + 1]$. \\
                Mula-mula dicari nilai harapan dan variansi dari distribusi marginal:
                \begin{align*}
                  E[X]   & = \sum_{x} x f(x) = 1(0.2) + 2(0.3) + 3(0.5) = 0.2 + 0.6 + 1.5 = 2.3         \\
                  E[X^2] & = \sum_{x} x^2 f(x) = 1^2(0.2) + 2^2(0.3) + 3^2(0.5) = 0.2 + 1.2 + 4.5 = 5.9 \\
                  Var[X] & = E[X^2] - (E[X])^2 = 5.9 - (2.3)^2 = 5.9 - 5.29 = 0.61
                \end{align*}
                Karena $X_1, X_2$ i.i.d, maka $E[X_1] = E[X_2] = 2.3$, $Var[X_1] = Var[X_2] = 0.61$, dan $X_1, X_2$ saling bebas sehingga $Cov(X_1, X_2) = 0$.
                \begin{align*}
                  E[2X_1 + 3X_2 + 1]   & = 2E[X_1] + 3E[X_2] + 1                              \\
                                       & = 2(2.3) + 3(2.3) + 1 = 4.6 + 6.9 + 1 = 12.5         \\
                  Var[2X_1 + 3X_2 + 1] & = 2^2 Var[X_1] + 3^2 Var[X_2] + 2(2)(3)Cov(X_1, X_2) \\
                                       & = 4(0.61) + 9(0.61) + 0                              \\
                                       & = 13(0.61) = 7.93
                \end{align*}
        \end{enumerate}

  \item Diketahui pdf bersama $f(x,y) = 1$ untuk $0 < y < 2x$ dan $0 < x < 1$. \\
        Batas daerah (support) dapat ditulis juga sebagai $0 < y < 2$ dan $y/2 < x < 1$.
        \begin{enumerate}[label=\alph*)]
          \item pdf marginal $f_1(x)$ dan $f_2(y)$:
                \begin{align*}
                  f_1(x) & = \int_{-\infty}^{\infty} f(x,y) \,dy = \int_{0}^{2x} 1 \,dy = [y]_{0}^{2x} = 2x, \quad \text{untuk } 0 < x < 1                \\
                  f_2(y) & = \int_{-\infty}^{\infty} f(x,y) \,dx = \int_{y/2}^{1} 1 \,dx = [x]_{y/2}^{1} = 1 - \frac{y}{2}, \quad \text{untuk } 0 < y < 2
                \end{align*}
          \item Dapatkan $P[Y \le 0.6 \mid X = 0.2]$. \\
                Pdf bersyarat $Y$ diberikan $X$:
                \[ f(y \mid x) = \frac{f(x,y)}{f_1(x)} = \frac{1}{2x}, \quad \text{untuk } 0 < y < 2x \]
                Untuk $X = 0.2$, nilai $Y$ berada pada interval $0 < y < 2(0.2) = 0.4$.
                \[ f(y \mid 0.2) = \frac{1}{2(0.2)} = \frac{1}{0.4} = 2.5, \quad \text{untuk } 0 < y < 0.4 \]
                Karena nilai maksimal $Y$ saat $X=0.2$ adalah $0.4$, maka $Y$ pasti kurang dari $0.6$.
                \begin{align*}
                  P[Y \le 0.6 \mid X = 0.2] & = \int_{-\infty}^{0.6} f(y \mid 0.2) \,dy                   \\
                                            & = \int_{0}^{0.4} 2.5 \,dy = [2.5y]_{0}^{0.4} = 2.5(0.4) = 1
                \end{align*}
          \item Dapatkan $E_X[E[y \mid x]]$. \\
                Berdasarkan hukum ekspektasi total (Law of Total Expectation), $E_X[E[Y \mid X]] = E[Y]$.
                \begin{align*}
                  E[Y] & = \int_{-\infty}^{\infty} y f_2(y) \,dy = \int_{0}^{2} y \left(1 - \frac{y}{2}\right) \,dy                \\
                       & = \int_{0}^{2} \left(y - \frac{y^2}{2}\right) \,dy = \left[ \frac{y^2}{2} - \frac{y^3}{6} \right]_{0}^{2} \\
                       & = \left( \frac{4}{2} - \frac{8}{6} \right) - 0 = 2 - \frac{4}{3} = \frac{2}{3}
                \end{align*}
                Cara alternatif (langsung):
                \begin{align*}
                  E[Y \mid x]      & = \int_{0}^{2x} y f(y \mid x) \,dy = \int_{0}^{2x} y \frac{1}{2x} \,dy = \frac{1}{2x} \left[ \frac{y^2}{2} \right]_{0}^{2x} = \frac{1}{2x} \left( \frac{4x^2}{2} \right) = x \\
                  E_X[E[Y \mid X]] & = E_X[X] = \int_{0}^{1} x f_1(x) \,dx = \int_{0}^{1} x(2x) \,dx = \int_{0}^{1} 2x^2 \,dx = \left[ \frac{2}{3}x^3 \right]_{0}^{1} = \frac{2}{3}
                \end{align*}
          \item Dapatkan $cov(X,Y)$.
                \begin{align*}
                  cov(X,Y) & = E[XY] - E[X]E[Y]                                                                                                                        \\
                  E[XY]    & = \int_{-\infty}^{\infty} \int_{-\infty}^{\infty} xy f(x,y) \,dy \,dx = \int_{0}^{1} \int_{0}^{2x} xy (1) \,dy \,dx                       \\
                           & = \int_{0}^{1} x \left[ \frac{y^2}{2} \right]_{0}^{2x} \,dx = \int_{0}^{1} x \left( \frac{4x^2}{2} \right) \,dx = \int_{0}^{1} 2x^3 \,dx  \\
                           & = \left[ \frac{2}{4}x^4 \right]_{0}^{1} = \frac{1}{2}                                                                                     \\
                  E[X]     & = \frac{2}{3} \quad \text{(dari perhitungan sebelumnya)}                                                                                  \\
                  E[Y]     & = \frac{2}{3} \quad \text{(dari perhitungan sebelumnya)}                                                                                  \\
                  cov(X,Y) & = \frac{1}{2} - \left(\frac{2}{3}\right)\left(\frac{2}{3}\right) = \frac{1}{2} - \frac{4}{9} = \frac{9}{18} - \frac{8}{18} = \frac{1}{18}
                \end{align*}
        \end{enumerate}

  \item Misalkan $X \sim WEI(\theta, \beta)$. Diketahui: \\
        $f(x) = \frac{\beta}{\theta^\beta} x^{\beta-1} e^{-\left(\frac{x}{\theta}\right)^\beta}, x > 0$ \\
        $F(x) = 1 - e^{-\left(\frac{x}{\theta}\right)^\beta}, x > 0$ \\
        Kita diminta mencari pdf dan CDF dari $Y = \ln X$.
        Karena $x > 0$, maka rentang nilai $Y = \ln X$ adalah $-\infty < y < \infty$. \\
        Untuk mendapatkan CDF dari $Y$, $F_Y(y)$:
        \begin{align*}
          F_Y(y) & = P(Y \le y)                                                          \\
                 & = P(\ln X \le y)                                                      \\
                 & = P(X \le e^y) \quad \text{(karena fungsi eksponensial monoton naik)} \\
                 & = F_X(e^y)                                                            \\
                 & = 1 - e^{-\left(\frac{e^y}{\theta}\right)^\beta}
        \end{align*}
        Jadi, CDF dari $Y$ adalah $F_Y(y) = 1 - e^{-\left(\frac{e^y}{\theta}\right)^\beta}$ untuk $-\infty < y < \infty$. \\
        Untuk mendapatkan pdf dari $Y$, $f_Y(y)$, kita turunkan $F_Y(y)$ terhadap $y$:
        \begin{align*}
          f_Y(y) & = \frac{d}{dy} F_Y(y)                                                                                                         \\
                 & = \frac{d}{dy} \left( 1 - \exp\left(-\left(\frac{e^y}{\theta}\right)^\beta\right) \right)                                     \\
                 & = - \exp\left(-\left(\frac{e^y}{\theta}\right)^\beta\right) \cdot \frac{d}{dy} \left(-\frac{e^{\beta y}}{\theta^\beta}\right) \\
                 & = - \exp\left(-\left(\frac{e^y}{\theta}\right)^\beta\right) \cdot \left(-\frac{\beta e^{\beta y}}{\theta^\beta}\right)        \\
                 & = \frac{\beta}{\theta^\beta} e^{\beta y} e^{-\left(\frac{e^y}{\theta}\right)^\beta}
        \end{align*}
        Jadi, pdf dari $Y$ adalah $f_Y(y) = \frac{\beta}{\theta^\beta} e^{\beta y} e^{-\left(\frac{e^y}{\theta}\right)^\beta}$ untuk $-\infty < y < \infty$.

  \item Misalkan pdf bersama $f(x,y) = e^{-y}$ untuk $0 < x < y < \infty$.
        \begin{enumerate}[label=\alph*)]
          \item Dapatkan pdf bersama dari $S = X + Y$ dan $T = X$. \\
                Diketahui transformasi $s = x + y$ dan $t = x$.
                Maka invers transformasinya adalah $x = t$ dan $y = s - t$. \\
                Jacobian transformasinya adalah:
                \[ J = \det \begin{pmatrix} \frac{\partial x}{\partial s} & \frac{\partial x}{\partial t} \\ \frac{\partial y}{\partial s} & \frac{\partial y}{\partial t} \end{pmatrix} = \det \begin{pmatrix} 0 & 1 \\ 1 & -1 \end{pmatrix} = (0)(-1) - (1)(1) = -1 \]
                Sehingga $|J| = |-1| = 1$. \\
                Menentukan daerah asal untuk $(s,t)$. Diketahui $0 < x < y < \infty$, maka:
                \[ 0 < t < s - t < \infty \]
                Dari $0 < t$ diperoleh $t > 0$. \\
                Dari $t < s - t$ diperoleh $2t < s \implies t < s/2$. Karena $t > 0$, maka $s > 0$. \\
                Sehingga rentangnya adalah $0 < t < s/2$ dan $s > 0$. \\
                Pdf bersama dari $S$ dan $T$ adalah:
                \begin{align*}
                  f_{S,T}(s,t) & = f_{X,Y}(x(s,t), y(s,t)) \cdot |J|                           \\
                               & = e^{-(s-t)} \cdot 1                                          \\
                               & = e^{t-s}, \quad \text{untuk } 0 < t < s/2 \text{ dan } s > 0
                \end{align*}
          \item Dapatkan pdf marginal dari $S$ dan $T$. \\
                Pdf marginal dari $S$:
                \begin{align*}
                  f_S(s) & = \int_{-\infty}^{\infty} f_{S,T}(s,t) \,dt = \int_{0}^{s/2} e^{t-s} \,dt = e^{-s} \int_{0}^{s/2} e^t \,dt \\
                         & = e^{-s} [e^t]_{0}^{s/2} = e^{-s} (e^{s/2} - e^0)                                                          \\
                         & = e^{-s/2} - e^{-s}, \quad \text{untuk } s > 0
                \end{align*}
                Pdf marginal dari $T$:
                \begin{align*}
                  f_T(t) & = \int_{-\infty}^{\infty} f_{S,T}(s,t) \,ds = \int_{2t}^{\infty} e^{t-s} \,ds = e^t \int_{2t}^{\infty} e^{-s} \,ds \\
                         & = e^t [-e^{-s}]_{2t}^{\infty} = e^t (0 - (-e^{-2t}))                                                               \\
                         & = e^t (e^{-2t})                                                                                                    \\
                         & = e^{-t}, \quad \text{untuk } t > 0
                \end{align*}
        \end{enumerate}
\end{enumerate}
