\begin{enumerate}
  \item \begin{enumerate}[label=\alph*.]
          \item Misalkan kejadian bola terambil merah $= M$, kuning $= K$, hijau $= H$. \\
                Maka ruang sampelnya adalah $S = \{M, K, H\}$.
          \item Semua kemungkinan kejadian adalah himpunan kuasa dari $S$ ($2^3=8$ kejadian), yaitu: \\
                $\Sigma = \{\emptyset, \{M\}, \{K\}, \{H\}, \{M, K\}, \{M, H\}, \{K, H\}, \{M, K, H\}\}$.
          \item Kejadian diperoleh dua bola dengan memperhatikan urutan: banyak anggota ruang sampel percobaan dua bola adalah $3 \times 3 = 9$. \\
                $S_2 = \{(M,M), (M,K), (M,H), (K,M), (K,K), (K,H), (H,M), (H,K), (H,H)\}$. \\
                Kejadian $A$ = paling sedikit terdapat satu bola merah = $\{(M,M), (M,K), (M,H), (K,M), (H,M)\}$. \\
                Asumsikan setiap warna memiliki peluang muncul yang sama, maka diperoleh $P(A) = \dfrac{n(A)}{n(S_2)} = \dfrac{5}{9}$.
          \item $W$ adalah kejadian munculnya selain bola merah dari kedua percobaan tersebut. Event $W$ adalah komplemen dari kejadian $A$. \\
                $W = A^c = \{(K,K), (K,H), (H,K), (H,H)\}$. \\
                Maka $P(W) = \dfrac{n(W)}{n(S_2)} = \dfrac{4}{9}$. Bisa juga dihitung dengan $P(W) = 1 - P(A) = 1 - \dfrac{5}{9} = \dfrac{4}{9}$.
        \end{enumerate}
        \vspace{0.3cm}

  \item Misalkan: \\
        $H_1$ = Kejadian terambil kartu hati dari tumpukan pertama. \\
        $H_1^c$ = Kejadian terambil bukan kartu hati dari tumpukan pertama. \\
        $H_2$ = Kejadian terambil kartu hati dari tumpukan kedua. \\
        Setiap tumpukan awalnya memiliki 52 kartu (dengan 13 kartu hati), maka $P(H_1) = \dfrac{13}{52} = \dfrac{1}{4}$ dan $P(H_1^c) = \dfrac{39}{52} = \dfrac{3}{4}$.

        \begin{enumerate}[label=\alph*.]
          \item Peluang kejadian kartu yang terambil dari tumpukan kedua adalah kartu hati ($H_2$): \\
                Jika $H_1$ terjadi, maka tumpukan kedua menjadi 53 kartu dengan 14 kartu hati. Jadi, $P(H_2|H_1) = \dfrac{14}{53}$. \\
                Jika $H_1^c$ terjadi, maka tumpukan kedua menjadi 53 kartu dengan 13 kartu hati. Jadi, $P(H_2|H_1^c) = \dfrac{13}{53}$. \\
                Berdasarkan Teorema Peluang Total:
                \begin{align*}
                  P(H_2) & = P(H_2|H_1)P(H_1) + P(H_2|H_1^c)P(H_1^c)                                                                     \\
                         & = \left(\dfrac{14}{53}\right)\left(\dfrac{1}{4}\right) + \left(\dfrac{13}{53}\right)\left(\dfrac{3}{4}\right) \\
                         & = \dfrac{14}{212} + \dfrac{39}{212} = \dfrac{53}{212} = \dfrac{1}{4}
                \end{align*}
          \item Peluang kejadian kartu hati yang terambil dari tumpukan kedua tersebut berasal dari tumpukan kartu pertama ($H_1|H_2$): \\
                Berdasarkan Teorema Bayes:
                \begin{align*}
                  P(H_1|H_2) & = \dfrac{P(H_2|H_1)P(H_1)}{P(H_2)}                          \\
                             & = \dfrac{\dfrac{14}{212}}{\dfrac{53}{212}} = \dfrac{14}{53}
                \end{align*}
        \end{enumerate}
        \vspace{0.3cm}

  \item Diketahui MGF $M_X(t) = \dfrac{5}{15}e^t + \dfrac{4}{15}e^{2t} + \dfrac{3}{15}e^{3t} + \dfrac{2}{15}e^{4t} + \dfrac{1}{15}e^{5t}$.
        \begin{enumerate}[label=\alph*.]
          \item Karena bentuk umum MGF peubah acak diskrit adalah $M_X(t) = \sum_{x} e^{tx} f_X(x)$, maka pdf (pmf) dari $X$ dapat ditentukan secara langsung:
                \[ f_X(x) = \begin{cases}
                    5/15, & x = 1                           \\
                    4/15, & x = 2                           \\
                    3/15, & x = 3                           \\
                    2/15, & x = 4                           \\
                    1/15, & x = 5                           \\
                    0,    & \text{untuk } x \text{ lainnya}
                  \end{cases} \]
          \item Untuk menghitung $E[3X+1]$ dan $Var[3X+1]$, terlebih dahulu dicari $E[X]$ dan $E[X^2]$:
                \begin{align*}
                  E[X]   & = \sum x \cdot f_X(x) = 1\left(\dfrac{5}{15}\right) + 2\left(\dfrac{4}{15}\right) + 3\left(\dfrac{3}{15}\right) + 4\left(\dfrac{2}{15}\right) + 5\left(\dfrac{1}{15}\right)             \\
                         & = \dfrac{5+8+9+8+5}{15} = \dfrac{35}{15} = \dfrac{7}{3}                                                                                                                                 \\
                  E[X^2] & = \sum x^2 \cdot f_X(x) = 1^2\left(\dfrac{5}{15}\right) + 2^2\left(\dfrac{4}{15}\right) + 3^2\left(\dfrac{3}{15}\right) + 4^2\left(\dfrac{2}{15}\right) + 5^2\left(\dfrac{1}{15}\right) \\
                         & = \dfrac{5+16+27+32+25}{15} = \dfrac{105}{15} = 7                                                                                                                                       \\
                  Var(X) & = E[X^2] - (E[X])^2 = 7 - \left(\dfrac{7}{3}\right)^2 = \dfrac{63}{9} - \dfrac{49}{9} = \dfrac{14}{9}
                \end{align*}
                Maka:
                \begin{align*}
                  E[3X+1]   & = 3E[X] + 1 = 3\left(\dfrac{7}{3}\right) + 1 = 7 + 1 = 8 \\
                  Var[3X+1] & = 3^2 Var(X) = 9\left(\dfrac{14}{9}\right) = 14
                \end{align*}
        \end{enumerate}
        \vspace{0.3cm}

  \item Diketahui $X$ terdistribusi eksponensial dengan: $f(x) = \dfrac{1}{\theta}e^{-x/\theta}, \quad x > 0; \, \theta > 0$.
        \begin{enumerate}[label=\alph*.]
          \item CDF dari $X$:
                \[ F_X(x) = \int_{-\infty}^{x} f(t) dt = \int_{0}^{x} \dfrac{1}{\theta}e^{-t/\theta} dt = \left[-e^{-t/\theta}\right]_{0}^{x} = 1 - e^{-x/\theta}, \quad \text{untuk } x > 0 \]
                Sehingga CDF adalah $F_X(x) = \begin{cases} 1 - e^{-x/\theta}, & x > 0 \\ 0, & x \le 0 \end{cases}$
          \item Nilai harapan $E[X]$:
                \[ E[X] = \int_{0}^{\infty} x \left(\dfrac{1}{\theta}e^{-x/\theta}\right) dx \]
                Dengan integrasi parsial atau mengingat ini adalah distribusi eksponensial berparameter $\beta = \theta$, didapatkan:
                \[ E[X] = \theta \]
          \item $P[X > 2]$:
                \[ P[X > 2] = 1 - P[X \le 2] = 1 - F_X(2) = 1 - (1 - e^{-2/\theta}) = e^{-2/\theta} \]
        \end{enumerate}
        \vspace{0.3cm}

  \item Peluang peluncuran misil sukses adalah $p = 0.9$ dan peluang gagal $q = 0.1$.
        \begin{enumerate}[label=\alph*.]
          \item $n = 10$ percobaan, misalkan peubah acak $A$ adalah banyak peluncuran misil yang sukses. Ini adalah kasus Distribusi Binomial, $A \sim Bin(n=10, p=0.9)$. \\
                Peluang paling sedikit 2 peluncuran sukses ($P(A \ge 2)$):
                \begin{align*}
                  P(A \ge 2) & = 1 - P(A < 2)                                                                        \\
                             & = 1 - \left( P(A=0) + P(A=1) \right)                                                  \\
                             & = 1 - \left( \binom{10}{0} (0.9)^0 (0.1)^{10} + \binom{10}{1} (0.9)^1 (0.1)^9 \right) \\
                             & = 1 - (0.1)^{10} - 9(0.1)^9                                                           \\
                             & = 1 - 9.1 \times 10^{-9} = 0.9999999909
                \end{align*}
          \item Uji dilakukan hingga diperoleh 3 kali sukses ($k=3$). Ini adalah kasus Distribusi Binomial Negatif. \\
                Misalkan peubah acak $Y$ adalah jumlah total percobaan peluncuran yang diperlukan untuk mendapat tepat sukses ke-3. ($Y \sim NB(k=3, p=0.9)$). \\
                Peluang tepat 6 kali peluncuran diperlukan ($Y=6$):
                \begin{align*}
                  P(Y = y) & = \binom{y-1}{k-1} p^k q^{y-k}         \\
                  P(Y = 6) & = \binom{6-1}{3-1} (0.9)^3 (0.1)^{6-3} \\
                           & = \binom{5}{2} (0.9)^3 (0.1)^3         \\
                           & = 10 (0.729) (0.001)                   \\
                           & = 0.00729
                \end{align*}
        \end{enumerate}
\end{enumerate}
