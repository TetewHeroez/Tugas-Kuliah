\begin{center}
  \textbf{SOAL}
\end{center}

\begin{enumerate}
  \item Hitung
        $\displaystyle
          \oint_C \frac{dz}{z^2 - (1+i)z + i}
        $
        dengan lintasan $C : |z| = 3$ berorientasi positif.

  \item Hitung residu dari fungsi
        $
          f(z) = \dfrac{z^2 - 2z}{(z+1)^2(z+4)}.
        $

  \item
        \begin{enumerate}
          \item Ekspansikan fungsi
                $\displaystyle
                  f(z) = \frac{1}{2+4z}
                $
                ke deret Maclaurin.
          \item Tentukan daerah konvergensi deret pangkat
                $\displaystyle
                  \sum_{k=0}^{\infty} e^k(z-2i)^k.
                $
        \end{enumerate}

  \item Nyatakan integral berikut ke bentuk Integral Cauchy dan selesaikan:
        \[
          \int_0^{2\pi} \left(\frac{\pi}{4} + 2e^{i\theta}\right)
          \sqrt{\tan\left(\frac{\pi}{4} + 2e^{i\theta}\right)}\, d\theta
        \]
\end{enumerate}

\begin{center}
  \textbf{SOLUSI}
\end{center}

\begin{enumerate}
  \item \jawab
        Faktorkan penyebut:
        \[
          z^2 - (1+i)z + i = (z-1)(z-i).
        \]
        Karena $z=1$ dan $z=i$ keduanya berada di dalam lintasan $|z|=3$, maka
        \[
          \oint_C \frac{dz}{(z-1)(z-i)}
          = 2\pi i \left( \operatorname{Res}_{z=1} f(z) + \operatorname{Res}_{z=i} f(z) \right).
        \]
        Residunya:
        \[
          \operatorname{Res}_{z=1} f(z) = \lim_{z \to 1} \frac{1}{z-i} = \frac{1}{1-i},
        \]
        \[
          \operatorname{Res}_{z=i} f(z) = \lim_{z \to i} \frac{1}{z-1} = \frac{1}{i-1} = -\frac{1}{1-i}.
        \]
        Jadi jumlah residunya nol, sehingga
        \[
          \oint_C \frac{dz}{z^2 - (1+i)z + i} = 0.
        \]

  \item \jawab
        Titik singular fungsi adalah $z=-1$ (pole orde $2$) dan $z=-4$ (pole sederhana).

        Untuk $z=-4$,
        \[
          \operatorname{Res}_{z=-4} f(z)
          = \lim_{z \to -4} (z+4)\frac{z^2-2z}{(z+1)^2(z+4)}
          = \frac{(-4)^2 - 2(-4)}{(-3)^2}
          = \frac{24}{9}
          = \frac{8}{3}.
        \]

        Untuk $z=-1$, gunakan rumus residu pada pole orde $2$:
        \[
          \operatorname{Res}_{z=-1} f(z)
          = \left.\frac{d}{dz}\left((z+1)^2 \frac{z^2-2z}{(z+1)^2(z+4)}\right)\right|_{z=-1}
          = \left.\frac{d}{dz}\left(\frac{z^2-2z}{z+4}\right)\right|_{z=-1}.
        \]
        Misalkan
        \[
          g(z) = \frac{z^2-2z}{z+4}.
        \]
        Maka
        \[
          g'(z) = \frac{(2z-2)(z+4) - (z^2-2z)}{(z+4)^2}.
        \]
        Substitusi $z=-1$ memberi
        \[
          \operatorname{Res}_{z=-1} f(z)
          = \frac{(-4)(3) - 3}{3^2}
          = \frac{-15}{9}
          = -\frac{5}{3}.
        \]

        Jadi,
        \[
          \operatorname{Res}_{z=-1} f(z) = -\frac{5}{3},
          \qquad
          \operatorname{Res}_{z=-4} f(z) = \frac{8}{3}.
        \]

  \item
        \begin{enumerate}[label=\alph*)]
          \item \jawab
                Tulis
                \[
                  \frac{1}{2+4z} = \frac{1}{2}\cdot \frac{1}{1+2z}.
                \]
                Dengan deret geometri
                \[
                  \frac{1}{1-w} = \sum_{n=0}^{\infty} w^n, \qquad |w|<1,
                \]
                ambil $w=-2z$, sehingga
                \[
                  \frac{1}{1+2z} = \sum_{n=0}^{\infty} (-2z)^n, \qquad |z| < \frac{1}{2}.
                \]
                Maka deret Maclaurin-nya adalah
                \[
                  \frac{1}{2+4z}
                  = \frac{1}{2}\sum_{n=0}^{\infty} (-2z)^n
                  = \sum_{n=0}^{\infty} (-1)^n 2^{n-1} z^n,
                  \qquad |z| < \frac{1}{2}.
                \]
                Beberapa suku awal:
                \[
                  \frac{1}{2+4z} = \frac{1}{2} - z + 2z^2 - 4z^3 + 8z^4 - \cdots
                \]

          \item \jawab
                Untuk deret
                \[
                  \sum_{k=0}^{\infty} e^k(z-2i)^k,
                \]
                tulis sebagai
                \[
                  \sum_{k=0}^{\infty} \bigl(e(z-2i)\bigr)^k.
                \]
                Ini adalah deret geometri dengan rasio
                \[
                  r=e(z-2i).
                \]
                Deret geometri konvergen jika dan hanya jika
                \[
                  |r|<1.
                \]
                Jadi
                \[
                  |e(z-2i)|<1
                  \iff
                  |z-2i|<\frac{1}{e}.
                \]
                Dengan demikian, daerah konvergensinya adalah
                \[
                  \{z\in\mathbb{C}: |z-2i|<\tfrac{1}{e}\},
                \]
                dengan jari-jari konvergensi
                \[
                  R=\frac{1}{e}.
                \]
        \end{enumerate}

  \item \jawab
        Ambil substitusi
        \[
          z = \frac{\pi}{4} + 2e^{i\theta}.
        \]
        Maka
        \[
          z - \frac{\pi}{4} = 2e^{i\theta}.
        \]
        Karena itu,
        \[
          d\theta = \frac{dz}{i\left(z-\frac{\pi}{4}\right)}.
        \]
        Integral berubah menjadi
        \[
          I
          =
          \oint_C
          \frac{z\sqrt{\tan z}}{i\left(z-\frac{\pi}{4}\right)}\,dz
          =
          \frac{1}{i}
          \oint_C
          \frac{z\sqrt{\tan z}}{z-\frac{\pi}{4}}\,dz,
        \]
        dengan $C$ adalah lingkaran
        \[
          \left|z-\frac{\pi}{4}\right|=2.
        \]
        Menggunakan Rumus Integral Cauchy,
        \[
          \oint_C \frac{f(z)}{z-a}\,dz = 2\pi i\,f(a),
        \]
        dengan
        \[
          f(z)=z\sqrt{\tan z},
          \qquad
          a=\frac{\pi}{4},
        \]
        diperoleh
        \[
          I
          =
          \frac{1}{i}
          \left(2\pi i\,f\left(\frac{\pi}{4}\right)\right)
          =
          2\pi \left(\frac{\pi}{4}\sqrt{\tan\frac{\pi}{4}}\right).
        \]
        Karena
        \[
          \tan\frac{\pi}{4}=1,
          \qquad
          \sqrt{1}=1,
        \]
        maka
        \[
          I = 2\pi \cdot \frac{\pi}{4} = \frac{\pi^2}{2}.
        \]
\end{enumerate}
