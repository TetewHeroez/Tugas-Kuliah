        untuk $0\leq t\leq 3$ adalah...\\
        \jawab
        \begin{flalign*}
            \int_{t_0}^{t_1}|F'(t)|\,dt&=\int_{0}^{3}\sqrt{(-3\sin{t})^2+(3\cos{t})^2+(3t^{1/2})^2}\,dt&\\
            &=\int_{0}^{3}\sqrt{9\sin^2{t}+9\cos^2{t}+9t}\,dt&\\
            &=\int_{0}^{3}\sqrt{9+9t}\,dt&\\
            &=9\int_{0}^{3}\sqrt{1+t}\,dt&\\
            &=9\left[\frac{2}{3}(1+t)^{3/2}\right]_0^3\,dt&\\
            &=6\left[(4)^{3/2}-1\right]_0^3\,dt&\\
            &=6\cdot 8=42
        \end{flalign*}
