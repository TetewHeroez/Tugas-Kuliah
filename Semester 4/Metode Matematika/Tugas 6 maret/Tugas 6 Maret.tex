                &=\frac{1}{(n^2+1)\pi}\left[e^{\pi}\cos(n\pi)-e^{-\pi}\cos(n\pi)\right]&\\
                &=\frac{e^{\pi}-e^{-\pi}}{(n^2+1)\pi}(-1)^n&\\
            \end{flalign*}
            \begin{flalign*}
