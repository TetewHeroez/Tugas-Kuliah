\begin{center}
  \underline{\textbf{SOAL}}
\end{center}

\begin{enumerate}[leftmargin=*, itemsep=0.35cm]
  \item Selesaikan PD dengan metode Deret Kuasa:
        $\displaystyle
          \frac{d^2y}{dt^2}-(t-1)y=0
        $
        di sekitar titik $t=1$.

  \item Selesaikan
        $\displaystyle
          \int_{-1}^{1} x^2P_4(x)\,dx.
        $
        Gunakan Rumus Rodrigues Polinomial Legendre.

  \item Dapatkan penyelesaian PD
        $\displaystyle
          (1-x^2)y''-2xy'+30y=0
        $
        dan tunjukkan bahwa penyelesaian yang didapat memenuhi PD tersebut.

  \item Jika diketahui
        $
          J_{\frac{1}{2}}(x)=\sqrt{\frac{2}{\pi x}}\sin x
        $ dan $
          J_{-\frac{1}{2}}(x)=\sqrt{\frac{2}{\pi x}}\cos x,
        $
        dapatkan:
        \begin{multicols}{2}
          \begin{enumerate}[label=\alph*)]
            \item $J_{\frac{3}{2}}(x)$
            \item $\displaystyle \int x^{5/2}J_{\frac{3}{2}}(x)\,dx$
          \end{enumerate}
        \end{multicols}
\end{enumerate}

\noindent \textbf{Catatan:} Rumus Rodrigues:
$\displaystyle
  P_n(x)=\frac{1}{2^n n!}\frac{d^n}{dx^n}(x^2-1)^n.
$

\begin{center}
  \textbf{SOLUSI}
\end{center}

\begin{enumerate}[leftmargin=*, itemsep=0.45cm]
  \item Untuk deret kuasa di sekitar $t=1$, ambil
        \[
          x=t-1.
        \]
        Maka PD menjadi
        \[
          \frac{d^2y}{dx^2}-xy=0.
        \]
        Misalkan solusi berbentuk
        \[
          y(x)=\sum_{n=0}^{\infty} a_nx^n.
        \]
        Maka
        \[
          y''(x)=\sum_{n=0}^{\infty}(n+2)(n+1)a_{n+2}x^n.
        \]
        Sementara itu,
        \[
          xy=\sum_{n=0}^{\infty}a_nx^{n+1}
          =\sum_{n=1}^{\infty}a_{n-1}x^n.
        \]
        Substitusi ke PD memberi
        \[
          \sum_{n=0}^{\infty}(n+2)(n+1)a_{n+2}x^n
          -
          \sum_{n=1}^{\infty}a_{n-1}x^n
          =0.
        \]
        Dari koefisien $x^0$ diperoleh
        \[
          2a_2=0 \quad \Longrightarrow \quad a_2=0.
        \]
        Untuk $n\ge 1$,
        \[
          (n+2)(n+1)a_{n+2}-a_{n-1}=0,
        \]
        sehingga rekurensinya
        \[
          a_{n+2}=\frac{a_{n-1}}{(n+2)(n+1)},\qquad n\ge 1.
        \]

        Ambil $a_0=C_1$ dan $a_1=C_2$. Maka
        \[
          a_2=0,\qquad
          a_3=\frac{a_0}{6},\qquad
          a_4=\frac{a_1}{12},\qquad
          a_5=0,
        \]
        \[
          a_6=\frac{a_3}{30}=\frac{a_0}{180},\qquad
          a_7=\frac{a_4}{42}=\frac{a_1}{504},\qquad
          a_8=0,
        \]
        dan seterusnya. Jadi
        \begin{align*}
          y(x)
           & =C_1\left(1+\frac{x^3}{6}+\frac{x^6}{180}+\frac{x^9}{12960}+\cdots\right)            \\
           & \quad +C_2\left(x+\frac{x^4}{12}+\frac{x^7}{504}+\frac{x^{10}}{45360}+\cdots\right).
        \end{align*}
        Karena $x=t-1$, penyelesaiannya adalah
        \begin{align*}
          y(t)
           & =C_1\left(1+\frac{(t-1)^3}{6}+\frac{(t-1)^6}{180}+\frac{(t-1)^9}{12960}+\cdots\right)                \\
           & \quad +C_2\left((t-1)+\frac{(t-1)^4}{12}+\frac{(t-1)^7}{504}+\frac{(t-1)^{10}}{45360}+\cdots\right).
        \end{align*}

  \item Dengan Rumus Rodrigues untuk $n=4$,
        \[
          P_4(x)=\frac{1}{2^4 4!}\frac{d^4}{dx^4}(x^2-1)^4
          =\frac{1}{384}\frac{d^4}{dx^4}(x^2-1)^4.
        \]
        Maka
        \[
          I:=\int_{-1}^{1}x^2P_4(x)\,dx
          =
          \frac{1}{384}\int_{-1}^{1}x^2\frac{d^4}{dx^4}(x^2-1)^4\,dx.
        \]
        Lakukan integrasi parsial empat kali. Karena $(x^2-1)^4=(x-1)^4(x+1)^4$,
        maka fungsi ini beserta turunan sampai orde $3$ bernilai nol di $x=\pm 1$.
        Jadi semua suku batas hilang, dan kita peroleh
        \[
          I
          =
          \frac{1}{384}\int_{-1}^{1}\frac{d^4}{dx^4}(x^2)\,(x^2-1)^4\,dx.
        \]
        Namun
        \[
          \frac{d^4}{dx^4}(x^2)=0.
        \]
        Akibatnya
        \[
          I=0.
        \]
        Jadi
        \[
          \boxed{\int_{-1}^{1}x^2P_4(x)\,dx=0.}
        \]

  \item Persamaan
        \[
          (1-x^2)y''-2xy'+30y=0
        \]
        adalah persamaan Legendre
        \[
          (1-x^2)y''-2xy'+n(n+1)y=0
        \]
        dengan
        \[
          n(n+1)=30.
        \]
        Jadi
        \[
          n=5.
        \]
        Karena itu solusi regularnya adalah polinomial Legendre orde $5$, yaitu
        \[
          y(x)=C\,P_5(x).
        \]
        Dari Rumus Rodrigues,
        \[
          P_5(x)=\frac{1}{2^5 5!}\frac{d^5}{dx^5}(x^2-1)^5
          =\frac{1}{8}(63x^5-70x^3+15x).
        \]
        Jadi salah satu penyelesaian adalah
        \[
          y(x)=C\left(\frac{1}{8}(63x^5-70x^3+15x)\right).
        \]
        Jika ingin ditulis lebih umum pada domain yang tidak memuat $\pm 1$, maka
        \[
          y(x)=C_1P_5(x)+C_2Q_5(x),
        \]
        dengan $Q_5$ fungsi Legendre jenis kedua.

        Sekarang kita verifikasi bahwa $P_5(x)$ memenuhi PD. Turunannya:
        \[
          P_5'(x)=\frac{1}{8}(315x^4-210x^2+15),
        \]
        \[
          P_5''(x)=\frac{1}{8}(1260x^3-420x).
        \]
        Maka
        \begin{align*}
          (1-x^2)P_5''(x)
           & =\frac{1}{8}(1260x^3-420x-1260x^5+420x^3) \\
           & =\frac{1}{8}(-1260x^5+1680x^3-420x),
        \end{align*}
        \[
          -2xP_5'(x)=\frac{1}{8}(-630x^5+420x^3-30x),
        \]
        \[
          30P_5(x)=\frac{1}{8}(1890x^5-2100x^3+450x).
        \]
        Jumlahkan:
        \begin{align*}
          (1-x^2)P_5''-2xP_5'+30P_5
           & =\frac{1}{8}\Bigl[
          (-1260-630+1890)x^5           \\
           & \qquad +(1680+420-2100)x^3 \\
           & \qquad +(-420-30+450)x
          \Bigr]                        \\
           & =0.
        \end{align*}
        Jadi benar bahwa $P_5(x)$ memenuhi persamaan diferensial tersebut.

  \item \begin{enumerate}[label=\alph*)]
          \item Gunakan rekurensi Bessel
                \[
                  J_{n-1}(x)+J_{n+1}(x)=\frac{2n}{x}J_n(x).
                \]
                Ambil $n=\frac{1}{2}$. Maka
                \[
                  J_{-\frac{1}{2}}(x)+J_{\frac{3}{2}}(x)=\frac{1}{x}J_{\frac{1}{2}}(x).
                \]
                Jadi
                \[
                  J_{\frac{3}{2}}(x)=\frac{1}{x}J_{\frac{1}{2}}(x)-J_{-\frac{1}{2}}(x).
                \]
                Substitusikan bentuk $J_{\pm 1/2}$:
                \begin{align*}
                  J_{\frac{3}{2}}(x)
                   & =\frac{1}{x}\sqrt{\frac{2}{\pi x}}\sin x-\sqrt{\frac{2}{\pi x}}\cos x \\
                   & =\sqrt{\frac{2}{\pi x}}\left(\frac{\sin x}{x}-\cos x\right).
                \end{align*}
                Jadi
                \[
                  \boxed{J_{\frac{3}{2}}(x)=\sqrt{\frac{2}{\pi x}}\left(\frac{\sin x}{x}-\cos x\right).}
                \]

          \item Dengan hasil butir (a),
                \begin{align*}
                  \int x^{5/2}J_{\frac{3}{2}}(x)\,dx
                   & =\int x^{5/2}\sqrt{\frac{2}{\pi x}}\left(\frac{\sin x}{x}-\cos x\right)\,dx \\
                   & =\sqrt{\frac{2}{\pi}}\int \left(x\sin x-x^2\cos x\right)\,dx.
                \end{align*}
                Hitung masing-masing integral:
                \[
                  \int x\sin x\,dx=-x\cos x+\sin x,
                \]
                \[
                  \int x^2\cos x\,dx=x^2\sin x+2x\cos x-2\sin x.
                \]
                Jadi
                \begin{align*}
                  \int \left(x\sin x-x^2\cos x\right)\,dx
                   & =(-x\cos x+\sin x)-(x^2\sin x+2x\cos x-2\sin x) \\
                   & =-x^2\sin x-3x\cos x+3\sin x.
                \end{align*}
                Maka
                \[
                  \boxed{
                    \int x^{5/2}J_{\frac{3}{2}}(x)\,dx
                    =
                    \sqrt{\frac{2}{\pi}}
                    \left(
                    -x^2\sin x-3x\cos x+3\sin x
                    \right)+C.
                  }
                \]
        \end{enumerate}
\end{enumerate}
