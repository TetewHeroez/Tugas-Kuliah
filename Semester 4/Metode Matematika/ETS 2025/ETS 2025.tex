  \textbf{SOLUSI}
\end{center}

\begin{enumerate}
  \item Untuk menyelesaikan $\int_{0}^{\infty} x^{1/4} e^{-\sqrt{x}} dx$, kita gunakan fungsi Gamma. \\
        Misalkan $u = \sqrt{x} = x^{1/2} \implies du = \frac{1}{2} x^{-1/2} dx \implies dx = 2u du$. \\
        Saat $x = 0$, $u = 0$. Saat $x \to \infty$, $u \to \infty$. \\
        Karena $x = u^2$, maka $x^{1/4} = u^{1/2}$. \\
        Substitusikan ke dalam integral:
        \begin{align*}
          \int_{0}^{\infty} u^{1/2} e^{-u} (2u du) & = 2 \int_{0}^{\infty} u^{3/2} e^{-u} du                                      \\
                                                   & = 2 \Gamma\left(\frac{5}{2}\right)                                           \\
                                                   & = 2 \cdot \frac{3}{2} \cdot \frac{1}{2} \cdot \Gamma\left(\frac{1}{2}\right) \\
                                                   & = \frac{3}{2} \sqrt{\pi}
        \end{align*}

  \item Untuk menentukan $\int_{0}^{3} \frac{x^2}{\sqrt{3-x}} dx$, kita gunakan fungsi Beta. \\
        Misalkan $x = 3t \implies dx = 3 dt$. \\
        Untuk batas integrasinya: saat $x = 0$, $t = 0$; saat $x = 3$, $t = 1$.
        \begin{align*}
          \int_{0}^{3} \frac{x^2}{\sqrt{3-x}} dx & = \int_{0}^{1} \frac{(3t)^2}{\sqrt{3-3t}} (3 dt)                                                   = \int_{0}^{1} \frac{9t^2}{\sqrt{3}\sqrt{1-t}} \cdot 3 dt                                          = \frac{27}{\sqrt{3}} \int_{0}^{1} t^2 (1-t)^{-1/2} dt                                                                                                                                                                                                                                                                                                                                                                                                      \\
                                                 & = 9\sqrt{3} \int_{0}^{1} t^{3-1} (1-t)^{1/2-1} dt                                                                                                                                                                                                            = 9\sqrt{3} \beta\left(3, \frac{1}{2}\right)                                                                                                                                                                                                                 = 9\sqrt{3} \frac{\Gamma(3) \Gamma(1/2)}{\Gamma(7/2)}                                                                                   \\
                                                 & = 9\sqrt{3} \frac{2! \cdot \sqrt{\pi}}{\frac{5}{2} \cdot \frac{3}{2} \cdot \frac{1}{2} \sqrt{\pi}}                                                                                                                                                                                                                                                                                                                                                                                                                                                                              = 9\sqrt{3} \frac{2}{15/8} = 9\sqrt{3} \cdot \frac{16}{15} = \frac{48\sqrt{3}}{5}
        \end{align*}

  \item \begin{enumerate}[label=\alph*)]
          \item Sketsa gelombang fungsi periodik $f(x) = x + \pi$, $-\pi < x < \pi$ untuk 3 periode.
                \begin{center}
                  \begin{tikzpicture}[scale=0.8]
                    % Sumbu x dan y
                    \draw[->] (-10.5,0) -- (10.5,0) node[right] {$x$};
                    \draw[->] (0,-1) -- (0,7) node[above] {$f(x)$};

                    % Garis asimtot y = 2\pi
                    \draw[dashed, gray] (-10.5, 6.28) -- (10.5, 6.28);

                    % Periode ketiga dari kiri (-3pi sampai -pi), \pi = 3.14
                    \draw[thick, blue] (-9.42, 0) -- (-3.14, 6.28);
                    \draw[dashed, blue] (-3.14, 6.28) -- (-3.14, 0);

                    % Periode utama (-pi sampai pi)
                    \draw[thick, blue] (-3.14, 0) -- (3.14, 6.28);
                    \draw[dashed, blue] (3.14, 6.28) -- (3.14, 0);

                    % Periode ketiga ke kanan (pi sampai 3pi)
                    \draw[thick, blue] (3.14, 0) -- (9.42, 6.28);
                    \draw[dashed, blue] (9.42, 6.28) -- (9.42, 0);

                    % Label koordinat di sumbu-x
                    \node[below] at (-9.42, 0) {$-3\pi$};
                    \node[below] at (-3.14, 0) {$-\pi$};
                    \node[below right] at (0, 0) {$0$};
                    \node[below] at (3.14, 0) {$\pi$};
                    \node[below] at (9.42, 0) {$3\pi$};

                    % Label koordinat di sumbu-y
                    \node[left] at (0, 6.28) {$2\pi$};
                  \end{tikzpicture}
                \end{center}
          \item Karena $f(-x) = -x + \pi \neq f(x)$ dan $f(-x) \neq -f(x)$, maka fungsi $f(x) = x + \pi$ di interval $(-\pi, \pi)$ adalah \textbf{bukan keduanya} (bukan fungsi genap dan bukan fungsi ganjil).
          \item Deret Fourier dari $f(x) = x + \pi$:
                $L = \pi$.
                \begin{align*}
                  a_0 & = \frac{1}{\pi} \int_{-\pi}^{\pi} (x + \pi) dx = \frac{1}{\pi} \left[ \frac{1}{2}x^2 + \pi x \right]_{-\pi}^{\pi} = \frac{1}{\pi} \left( (\frac{1}{2}\pi^2 + \pi^2) - (\frac{1}{2}\pi^2 - \pi^2) \right) = \frac{1}{\pi} (2\pi^2) = 2\pi \\
                  a_n & = \frac{1}{\pi} \int_{-\pi}^{\pi} (x + \pi) \cos(nx) dx = 0                                                                                                                                                                              \\
                  b_n & = \frac{1}{\pi} \int_{-\pi}^{\pi} (x + \pi) \sin(nx) dx = \frac{1}{\pi} \left( \left[ \frac{-(x+\pi)}{n}\cos(nx) \right]_{-\pi}^{\pi} - \int_{-\pi}^{\pi} \frac{-\cos(nx)}{n} dx \right)                                                 \\
                      & = \frac{1}{\pi} \left( \frac{-2\pi}{n}\cos(n\pi) - 0 + 0 \right) = -\frac{2}{n}(-1)^n = 2\frac{(-1)^{n+1}}{n}
                \end{align*}
                Sehingga deret Fouriernya:
                \[ f(x) = \pi + \sum_{n=1}^{\infty} 2\frac{(-1)^{n+1}}{n} \sin(nx) \]
        \end{enumerate}

  \item \begin{enumerate}[label=\alph*.]
          \item Tabel Beda Maju:
                \begin{center}
                  \begin{tabular}{|c|c|c|c|c|c|}
                    \hline
                    $x$ & $f(x)$ & $\Delta f(x)$ & $\Delta^2 f(x)$ & $\Delta^3 f(x)$ & $\Delta^4 f(x)$ \\
                    \hline
                    0   & 1      &               &                 &                 &                 \\
                        &        & 1             &                 &                 &                 \\
                    1   & 2      &               & 1               &                 &                 \\
                        &        & 2             &                 & 1               &                 \\
                    2   & 4      &               & 2               &                 & 1               \\
                        &        & 4             &                 & 2               &                 \\
                    3   & 8      &               & 4               &                 &                 \\
                        &        & 8             &                 &                 &                 \\
                    4   & 16     &               &                 &                 &                 \\
                    \hline
                  \end{tabular}
                \end{center}
          \item Berdasarkan tabel interpolasi Newton maju dengan titik dasar $x_0 = 0, x=2.5$ dan langkah $h=1 \implies s = \frac{x-x_0}{h} = \frac{2.5-0}{1} = 2.5$.
                Nilai beda pertamanya masing-masing dari $x_0=0$ adalah:
                $f(0)=1, \Delta f(0)=1, \Delta^2 f(0)=1, \Delta^3 f(0)=1, \Delta^4 f(0)=1$.

                Maka dengan rumus Newton Maju:
                \begin{align*}
                  f(2.5) & \approx f(0) + s\Delta f(0) + \frac{s(s-1)}{2!}\Delta^2 f(0) + \frac{s(s-1)(s-2)}{3!}\Delta^3 f(0) + \frac{s(s-1)(s-2)(s-3)}{4!}\Delta^4 f(0) \\
                         & \approx 1 + 2.5(1) + \frac{(2.5)(1.5)}{2}(1) + \frac{(2.5)(1.5)(0.5)}{6}(1) + \frac{(2.5)(1.5)(0.5)(-0.5)}{24}(1)                             \\
                         & \approx 1 + 2.5 + 1.875 + 0.3125 - 0.0390625                                                                                                  \\
                         & \approx 5.6484375
                \end{align*}
          \item Berdasarkan data tabel: $f(x)$ selalu berbentuk fungsi eksponensial $f(x) = 2^x$. Hal ini juga terlihat dari nilai karakteristik kolom perbedaan yang selalu konstan secara berturut ($\Delta^k f(x) = f(x)$ untuk interval langkah $h=1$).
        \end{enumerate}

  \item Persamaan Beda: $y_{n+2} - 3y_{n+1} + 2y_n = 1$, dengan $y_0 = 1, y_1 = 0$. \\
        Untuk solusi homogen $y_h(n)$:
        \begin{align*}
          (E^2 - 3E + 2) y_n & = 0 \\
          (E-1)(E-2) y_n     & = 0
        \end{align*}
        Sehingga akar karakteristiknya $E_1 = 1, E_2 = 2$. \\
        Maka $y_h(n) = C_1 (1)^n + C_2 (2)^n = C_1 + C_2 2^n$. \\

        Sedangkan untuk mencari solusi partikulir $y_p(n)$, gunakan pendekatan operator translasi ($E$):
        \begin{align*}
          y_p(n) & = \frac{1}{E^2 - 3E + 2} (1^n) \\
                 & = \frac{1}{(E-1)(E-2)} (1^n)
        \end{align*}
        Karena jika kita langsung mensubstitusikan $E = 1$ nilai pembaginya $0$, kita akan modifikasi fraksi $(E-1)$ tersebut menggunakan sifat Operator Beda $(\Delta = E - 1)$:
        \begin{align*}
          y_p(n) & = \frac{1}{E-1} \left( \frac{1}{1-2} 1^n \right) = \frac{1}{\Delta} (-1)
        \end{align*}
        Nilai $\frac{1}{\Delta}(1)$ adalah antiturun diskrit dari 1 (sebagaimana dengan deret), yang nilainya adalah $n$, sehingga didapatkan:
        \[ y_p(n) = -n \]

        Solusi umum: $y_n = y_h(n) + y_p(n) = C_1 + C_2 2^n - n$.
        Masukkan syarat awal:
        \begin{align*}
          y_0 & = 1 \implies C_1 + C_2 - 0 = 1 \implies C_1 + C_2 = 1   \\
          y_1 & = 0 \implies C_1 + 2C_2 - 1 = 0 \implies C_1 + 2C_2 = 1
        \end{align*}
        Dengan mengurangkan kedua persamaan: $(C_1+2C_2) - (C_1+C_2) = 1 - 1 \implies C_2 = 0$. \\
        Sehingga $C_1 = 1$. \\
        Maka Solusi dari persamaan beda tersebut adalah:
        \[ y_n = 1 - n \]
\end{enumerate}
