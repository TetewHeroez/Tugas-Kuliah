\begin{enumerate}
  \item Diketahui fungsi $f(x) = \sin(x)$ pada interval $[0,\pi]$. Jumlah Riemann dirumuskan sebagai $S(f, \mathcal{P}) = \sum_{i=1}^{n} f(t_i) \Delta x_i$.
        \begin{enumerate}[label=\alph*)]
          \item Partisi $\mathcal{P}_1 = \left\{0, \frac{\pi}{3}, \frac{3\pi}{4}, \pi\right\}$ dengan tanda di titik kiri ($t_i = x_{i-1}$).
                \begin{itemize}
                  \item Subinterval 1: $\left[0, \frac{\pi}{3}\right]$, $\Delta x_1 = \frac{\pi}{3}$, $t_1 = 0 \implies \sin(0) = 0$
                  \item Subinterval 2: $\left[\frac{\pi}{3}, \frac{3\pi}{4}\right]$, $\Delta x_2 = \frac{3\pi}{4} - \frac{\pi}{3} = \frac{9\pi}{12} - \frac{4\pi}{12} = \frac{5\pi}{12}$, $t_2 = \frac{\pi}{3} \implies \sin\left(\frac{\pi}{3}\right) = \frac{\sqrt{3}}{2}$
                  \item Subinterval 3: $\left[\frac{3\pi}{4}, \pi\right]$, $\Delta x_3 = \pi - \frac{3\pi}{4} = \frac{\pi}{4}$, $t_3 = \frac{3\pi}{4} \implies \sin\left(\frac{3\pi}{4}\right) = \frac{\sqrt{2}}{2}$
                \end{itemize}
                \begin{align*}
                  S(f, \mathcal{P}_1) & = \sin(0)\left(\frac{\pi}{3}\right) + \sin\left(\frac{\pi}{3}\right)\left(\frac{5\pi}{12}\right) + \sin\left(\frac{3\pi}{4}\right)\left(\frac{\pi}{4}\right) \\
                                      & = 0 + \left(\frac{\sqrt{3}}{2}\right)\left(\frac{5\pi}{12}\right) + \left(\frac{\sqrt{2}}{2}\right)\left(\frac{\pi}{4}\right)                                \\
                                      & = \frac{5\pi\sqrt{3}}{24} + \frac{\pi\sqrt{2}}{8}
                \end{align*}

          \item Partisi $\mathcal{P}_2 = \left\{0, \frac{\pi}{4}, \frac{5\pi}{6}, \pi\right\}$ dengan tanda di titik kanan ($t_i = x_i$).
                \begin{itemize}
                  \item Subinterval 1: $\left[0, \frac{\pi}{4}\right]$, $\Delta x_1 = \frac{\pi}{4}$, $t_1 = \frac{\pi}{4} \implies \sin\left(\frac{\pi}{4}\right) = \frac{\sqrt{2}}{2}$
                  \item Subinterval 2: $\left[\frac{\pi}{4}, \frac{5\pi}{6}\right]$, $\Delta x_2 = \frac{5\pi}{6} - \frac{\pi}{4} = \frac{10\pi}{12} - \frac{3\pi}{12} = \frac{7\pi}{12}$, $t_2 = \frac{5\pi}{6} \implies \sin\left(\frac{5\pi}{6}\right) = \frac{1}{2}$
                  \item Subinterval 3: $\left[\frac{5\pi}{6}, \pi\right]$, $\Delta x_3 = \pi - \frac{5\pi}{6} = \frac{\pi}{6}$, $t_3 = \pi \implies \sin(\pi) = 0$
                \end{itemize}
                \begin{align*}
                  S(f, \mathcal{P}_2) & = \sin\left(\frac{\pi}{4}\right)\left(\frac{\pi}{4}\right) + \sin\left(\frac{5\pi}{6}\right)\left(\frac{7\pi}{12}\right) + \sin(\pi)\left(\frac{\pi}{6}\right) \\
                                      & = \left(\frac{\sqrt{2}}{2}\right)\left(\frac{\pi}{4}\right) + \left(\frac{1}{2}\right)\left(\frac{7\pi}{12}\right) + 0                                         \\
                                      & = \frac{\pi\sqrt{2}}{8} + \frac{7\pi}{24}
                \end{align*}

          \item Partisi $\mathcal{P}_3 = \left\{0, \frac{\pi}{3}, \frac{2\pi}{3}, \pi\right\}$ dengan tanda di titik tengah ($t_i = \frac{x_{i-1} + x_i}{2}$).
                \begin{itemize}
                  \item Subinterval 1: $\left[0, \frac{\pi}{3}\right]$, $\Delta x_1 = \frac{\pi}{3}$, $t_1 = \frac{0 + \frac{\pi}{3}}{2} = \frac{\pi}{6} \implies \sin\left(\frac{\pi}{6}\right) = \frac{1}{2}$
                  \item Subinterval 2: $\left[\frac{\pi}{3}, \frac{2\pi}{3}\right]$, $\Delta x_2 = \frac{2\pi}{3} - \frac{\pi}{3} = \frac{\pi}{3}$, $t_2 = \frac{\frac{\pi}{3} + \frac{2\pi}{3}}{2} = \frac{\pi}{2} \implies \sin\left(\frac{\pi}{2}\right) = 1$
                  \item Subinterval 3: $\left[\frac{2\pi}{3}, \pi\right]$, $\Delta x_3 = \pi - \frac{2\pi}{3} = \frac{\pi}{3}$, $t_3 = \frac{\frac{2\pi}{3} + \pi}{2} = \frac{5\pi}{6} \implies \sin\left(\frac{5\pi}{6}\right) = \frac{1}{2}$
                \end{itemize}
                \begin{align*}
                  S(f, \mathcal{P}_3) & = \sin\left(\frac{\pi}{6}\right)\left(\frac{\pi}{3}\right) + \sin\left(\frac{\pi}{2}\right)\left(\frac{\pi}{3}\right) + \sin\left(\frac{5\pi}{6}\right)\left(\frac{\pi}{3}\right) \\
                                      & = \left(\frac{1}{2}\right)\left(\frac{\pi}{3}\right) + (1)\left(\frac{\pi}{3}\right) + \left(\frac{1}{2}\right)\left(\frac{\pi}{3}\right)                                         \\
                                      & = \frac{\pi}{6} + \frac{\pi}{3} + \frac{\pi}{6} = \frac{2\pi}{3}
                \end{align*}
        \end{enumerate}

  \item Misalkan $P = \{x_0, x_1, \dots, x_n\}$ adalah sembarang partisi dari interval $I = [a,b]$. Karena $f_1$ dan $f_2$ adalah fungsi yang didefinisikan pada subinterval $I_k = [x_{k-1}, x_k]$, kita memiliki sifat infimum untuk sembarang penjumlahan fungsi:
        \[ \inf_{x \in I_k} (f_1(x) + f_2(x)) \geq \inf_{x \in I_k} f_1(x) + \inf_{x \in I_k} f_2(x) \]
        (\textit{Bukti:} Berdasarkan definisi infimum, untuk setiap $x \in I_k$ berlaku $f_1(x) \geq \inf_{x \in I_k} f_1(x)$ dan $f_2(x) \geq \inf_{x \in I_k} f_2(x)$. Jika dijumlahkan, diperoleh $f_1(x) + f_2(x) \geq \inf_{x \in I_k} f_1(x) + \inf_{x \in I_k} f_2(x)$ untuk setiap $x \in I_k$. Hal ini berarti $\inf_{x \in I_k} f_1(x) + \inf_{x \in I_k} f_2(x)$ adalah batas bawah dari himpunan $\{f_1(x) + f_2(x) \mid x \in I_k\}$. Karena $\inf_{x \in I_k} (f_1(x) + f_2(x))$ adalah batas bawah terbesar dari himpunan tersebut, maka nilainya pasti lebih besar atau sama dengan sembarang batas bawah, sehingga sifat tersebut terbukti.) \\
        Atau dengan menggunakan notasi $m_k$, berlaku:
        \[ m_k(f_1 + f_2) \geq m_k(f_1) + m_k(f_2) \]
        Kalikan kedua ruas dengan $\Delta x_k$ dan jumlahkan dari $k=1$ sampai $n$:
        \[ \sum_{k=1}^n m_k(f_1 + f_2) \Delta x_k \geq \sum_{k=1}^n m_k(f_1) \Delta x_k + \sum_{k=1}^n m_k(f_2) \Delta x_k \]
        Sehingga untuk sembarang partisi $P$, berlaku pada jumlah Darboux bawahnya:
        \[ L(f_1 + f_2, P) \geq L(f_1, P) + L(f_2, P) \]
        Misalkan diberikan sembarang dua partisi $P_1$ dan $P_2$. Tinjau partisi gabungan $P^* = P_1 \cup P_2$. Karena $P^*$ adalah penghalusan (refinement) dari $P_1$ dan $P_2$, maka jumlah Darboux bawah selalu meningkat atau sama dengan partisi asalnya:
        \[ L(f_1, P_1) \leq L(f_1, P^*) \quad \text{dan} \quad L(f_2, P_2) \leq L(f_2, P^*) \]
        Sehingga:
        \[ L(f_1, P_1) + L(f_2, P_2) \leq L(f_1, P^*) + L(f_2, P^*) \leq L(f_1 + f_2, P^*) \]
        Karena nilai integral batas bawah didefinisikan sebagai supremum keseluruhan partisi, maka $L(f_1 + f_2, P^*) \leq L(f_1 + f_2)$, yang berimplikasi:
        \[ L(f_1, P_1) + L(f_2, P_2) \leq L(f_1 + f_2) \]
        Ambil supremum untuk semua kemungkinan partisi $P_1$:
        \[ L(f_1) + L(f_2, P_2) \leq L(f_1 + f_2) \]
        Akhirnya, ambil supremum untuk semua kemungkinan partisi $P_2$:
        \[ L(f_1) + L(f_2) \leq L(f_1 + f_2) \]
        Terbukti.

  \item Diberikan barisan fungsi $f_n(x) = \frac{2025nx}{1+2025n^2x^2}$ untuk $x \in \mathbb{R}$.
        \begin{enumerate}[label=\alph*)]
          \item Evaluasi limit \textit{pointwise}:
                \begin{itemize}
                  \item Jika $x = 0$, maka $f_n(0) = 0$ untuk setiap $n$, maka $\lim_{n \to \infty} f_n(0) = 0$.
                  \item Jika $x \neq 0$, bagilah pembilang dan penyebut dengan $n^2$:
                        \[ \lim_{n \to \infty} f_n(x) = \lim_{n \to \infty} \frac{2025 \cdot \frac{x}{n}}{\frac{1}{n^2} + 2025x^2} = \frac{0}{0 + 2025x^2} = 0 \]
                \end{itemize}
                Oleh karena itu, fungsi limitnya adalah $f(x) = 0$ secara \textit{pointwise} untuk semua $x \in \mathbb{R}$.
          \item Pada interval $[1, \infty)$, akan diselidiki konvergensi seragam.
                Cari titik maksimum dari $|f_n(x)|$ menggunakan turunan pertama:
                \[ f_n'(x) = \frac{2025n(1+2025n^2x^2) - 2025nx(4050n^2x)}{(1+2025n^2x^2)^2} = \frac{2025n(1 - 2025n^2x^2)}{(1+2025n^2x^2)^2} \]
                Titik kritis dicapai saat $f_n'(x) = 0$, yaitu:
                \[ 1 - 2025n^2x^2 = 0 \implies x^2 = \frac{1}{2025n^2} \implies x = \frac{1}{45n} \]
                Namun, area yang ditinjau pada interval ini adalah $x \in [1, \infty)$.
                Karena untuk setiap $n \ge 1$ titik ekstrem $x = \frac{1}{45n} \le \frac{1}{45} < 1$, maka titik tersebut berada di luar interval $[1, \infty)$.
                Selanjutnya, untuk setiap $x \ge 1$, nilai $1 - 2025n^2x^2 < 0$, sehingga $f_n'(x) < 0$. Hal ini menunjukkan bahwa $f_n(x)$ adalah fungsi yang monoton turun pada interval $[1, \infty)$.
                Oleh karena itu, jarak interval deviasinya (supremum norm) dicapai pada batas asimtot terdekat kurvanya di ujung kiri yakni $x = 1$:
                \[ M_n = \sup_{x \geq 1} |f_n(x) - f(x)| = \sup_{x \geq 1} \frac{2025nx}{1+2025n^2x^2} = f_n(1) = \frac{2025n}{1+2025n^2} \]
                Ambil nilai limit $n \to \infty$:
                \[ \lim_{n \to \infty} M_n = \lim_{n \to \infty} \frac{2025n}{1+2025n^2} = \lim_{n \to \infty} \frac{\frac{2025}{n}}{\frac{1}{n^2} + 2025} = \frac{0}{0 + 2025} = 0 \]
                Karena barisan norm supremumnya konvergen menuju 0 ($\lim_{n \to \infty} M_n = 0$), maka terbukti bahwa barisan fungsi $(f_n)$ tersebut konvergen seragam ke fungsi $f(x)=0$ pada rentang interval $[1,\infty)$.
        \end{enumerate}

  \item Akan dibuktikan secara langsung menggunakan Teorema Nilai Ekstrim (\textit{Extreme Value Theorem}).
        Diketahui bahwa $h$ adalah fungsi kontinu pada interval tertutup dan terbatas $[a,b]$. Menurut Teorema Nilai Ekstrim, fungsi $h$ pasti mencapai nilai minimum global pada selang tersebut. Artinya, terdapat suatu titik $c \in [a,b]$ sedemikian sehingga untuk setiap $x \in [a,b]$ berlaku:
        \[ h(x) \geq h(c) \]
        Karena disyaratkan dalam soal bahwa $h(x) > 0$ untuk setiap $x \in [a,b]$, maka nilai dari eksistensi titik minimum tersebut juga harus bernilai positif:
        \[ h(c) > 0 \]
        Misalkan $m = h(c)$, maka akan diperoleh suatu batas bawah konstan $h(x) \geq m > 0$ untuk semua nilai $x \in [a,b]$.
        Berdasarkan sifat keterurutan dari integral (kemonotonan integral tentu), berlaku:
        \[ \int_a^b h(x) \, dx \geq \int_a^b m \, dx \]
        Dengan mengevaluasi hasil integral fungsi bernilai konstan di ruas kanan tersebut, didapatkan nilai luas panjang interval batasan:
        \[ \int_a^b m \, dx = m(b - a) \]
        Karena kita bekerja pada rentang interval garis riil $[a,b]$, disyaratkan bahwa $b > a$, yang setara dengan menunjukan bahwa lebar selangnya $(b - a) > 0$. Diketahui pula dari konstruksi sebelumnya bahwa batas minimumnya bernilai $m > 0$. Titik pertemuan antara dua himpunan bilangan real dengan tanda positif, hasil kalinya pastilah lebih besar mutlak dari nol:
        \[ m(b - a) > 0 \]
        Berdasarkan berlakunya sifat transitif ketaksamaan pada gabungan integralnya, maka kita susun utuh menjadi batasan:
        \[ \int_a^b h(x) \, dx \ge m(b - a) > 0 \implies \int_a^b h(x) \, dx > 0 \]
        Ketaksamaan berkonjungsi tersebut menyimpulkan terbuktinya nilai luasan integral totalnya selalu bernilai bilangan positif tanpa bisa bernilai nol. Terbukti.
\end{enumerate}
