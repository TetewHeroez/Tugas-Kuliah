\begin{enumerate}
  \item Diberikan barisan fungsi $(f_n)$ dengan
        \[
          f_n(x) = \frac{x^n}{n}, \qquad x \in [0,1].
        \]
        \begin{enumerate}
          \item Tunjukkan bahwa $(f_n)$ konvergen seragam ke suatu fungsi $f$ pada $[0,1]$.
          \item Dapatkan $f_n'(x)$, $f'(x)$, dan
                $
                  \lim_{n\to\infty} f_n'(x).
                $
                Jika
                $
                  g(x) := \lim_{n\to\infty} f_n'(x),
                $
                tunjukkan apakah $g(1)=f'(1)$.
        \end{enumerate}

  \item Diberikan deret fungsi
        $
          \sum_{n=1}^{\infty} f_n,
        $ dengan
        $
          f_n(x) = \sin\left(\frac{x}{n^2}\right).
        $
        Apakah deret tersebut konvergen seragam pada $[0,\pi]$? Jelaskan.

  \item Tunjukkan bahwa interval $[1,2)$ tidak kompak.

  \item Diberikan $X=\R^2$. Didefinisikan fungsi $\rho : X\times X \to \R$ dengan
        $
          \rho(P_1,P_2) := |x_1-x_2|
        $
        untuk setiap $P_1(x_1,y_1),P_2(x_2,y_2)\in\R^2$. Apakah $\rho$ merupakan metrik di $\R^2$? Buktikan atau beri contoh penyangkal.
\end{enumerate}

  \textbf{SOLUSI}
\end{center}

\begin{enumerate}
  \item
        \begin{enumerate}[label=\alph*)]
          \item \jawab
                Untuk setiap $x\in[0,1]$,
                \[
                  f_n(x)=\frac{x^n}{n}.
                \]
                Karena $0\le x\le 1$, maka $0\le x^n\le 1$. Jadi
                \[
                  0\le \frac{x^n}{n}\le \frac{1}{n}.
                \]
                Dari sini langsung diperoleh, untuk setiap $x\in[0,1]$,
                \[
                  f_n(x)\to 0.
                \]
                Jadi fungsi limitnya adalah
                \[
                  f(x)=0.
                \]

                Sekarang cek konvergensi uniform:
                \[
                  \sup_{x\in[0,1]} |f_n(x)-0|
                  =
                  \sup_{x\in[0,1]} \frac{x^n}{n}.
                \]
                Karena $x^n\le 1$ untuk $x\in[0,1]$, maka supremum dicapai di $x=1$, sehingga
                \[
                  \sup_{x\in[0,1]} \frac{x^n}{n} = \frac{1}{n}\to 0.
                \]
                maka $f_n\to f$ secara uniform pada $[0,1]$.

          \item \jawab
                Karena
                \[
                  f_n(x)=\frac{x^n}{n},
                \]
                maka
                \[
                  f_n'(x)=x^{n-1}.
                \]

                Dari bagian (a), fungsi limitnya adalah $f(x)=0$, sehingga
                \[
                  f'(x)=0.
                \]

                Sekarang tinjau limit turunan:
                \[
                  \lim_{n\to\infty} f_n'(x)
                  =
                  \lim_{n\to\infty} x^{n-1}.
                \]
                Jika $0\le x<1$, maka
                \[
                  x^{n-1}\to 0.
                \]
                Sedangkan untuk $x=1$,
                \[
                  1^{n-1}=1.
                \]
                Jadi
                \[
                  g(x)=\lim_{n\to\infty} f_n'(x)
                  =
                  \begin{cases}
                    0, & 0\le x<1, \\
                    1, & x=1.
                  \end{cases}
                \]

                Khususnya,
                \[
                  g(1)=1,
                  \qquad
                  f'(1)=0.
                \]
                Jadi
                \[
                  g(1)\ne f'(1).
                \]
                Ini menunjukkan bahwa meskipun $f_n\to f$ secara uniform, limit turunan tidak selalu sama dengan turunan fungsi limit.
        \end{enumerate}

  \item \jawab
        Kita gunakan Weierstrass M-Test. Untuk $x\in[0,\pi]$ berlaku
        \[
          \left|\sin\left(\frac{x}{n^2}\right)\right|
          \le
          \frac{x}{n^2}
          \le
          \frac{\pi}{n^2}.
        \]
        Ambil
        \[
          M_n=\frac{\pi}{n^2}.
        \]
        Karena deret
        \[
          \sum_{n=1}^{\infty} M_n
          =
          \pi\sum_{n=1}^{\infty}\frac{1}{n^2}
        \]
        konvergen, maka menurut Weierstrass M-Test, deret
        \[
          \sum_{n=1}^{\infty}\sin\left(\frac{x}{n^2}\right)
        \]
        konvergen seragam pada $[0,\pi]$.

        Karena setiap suku adalah fungsi kontinu dan deretnya konvergen seragam, maka fungsi jumlahnya juga kontinu pada $[0,\pi]$.

  \item \jawab
        Menurut Teorema Heine-Borel, suatu himpunan di $\R$ kompak jika dan hanya jika tertutup dan terbatas.

        Himpunan $[1,2)$ memang terbatas, tetapi tidak tertutup di $\R$ karena titik $2$ adalah titik limit dari $[1,2)$ namun $2\notin[1,2)$.
        Maka $[1,2)$ tidak kompak.

  \item \jawab
        Fungsi
        \[
          \rho(P_1,P_2)=|x_1-x_2|
        \]
        tidak merupakan metrik pada $\R^2$.

        Memang, syarat positif dan simetri terpenuhi, tetapi syarat
        \[
          \rho(P_1,P_2)=0 \iff P_1=P_2
        \]
        gagal.

        Ambil contoh
        \[
          P_1=(0,0), \qquad P_2=(0,1).
        \]
        Jelas $P_1\neq P_2$, tetapi
        \[
          \rho(P_1,P_2)=|0-0|=0.
        \]
        Jadi terdapat dua titik berbeda yang berjarak nol.

        Dengan demikian, $\rho$ bukan metrik pada $\R^2$.
\end{enumerate}
