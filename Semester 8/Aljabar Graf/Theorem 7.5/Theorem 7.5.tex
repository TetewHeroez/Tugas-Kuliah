\begin{frame}{Definisi Dasar}
  \begin{itemize}
    \item \textbf{Hutan (Forest - $\Phi$):} Subgraf dari $\Gamma$ yang tidak mengandung siklus. Setiap komponen terhubung dalam hutan adalah sebuah pohon (tree).
    \item \textbf{Fungsi $p(\Phi)$:} Jumlah cara menghilangkan tepat satu verteks dari setiap komponen $\Phi$. Jika komponen $\Phi$ adalah $C_1, C_2, \dots, C_k$, maka $p(\Phi) = \prod_{j=1}^k |V(C_j)|$.
    \item \textbf{Matriks Insidensi Berarah ($\mathbf{D}$):}
          Matriks ukuran $|V| \times |E|$ dengan entri $\mathbf{D}_{v,e}$ bernilai $1$ jika $v$ adalah ujung kepala dari $e$, $-1$ jika $v$ adalah ujung ekor dari $e$, dan $0$ jika $v$ tidak insiden dengan $e$.
    \item \textbf{Matriks Derajat ($\Delta$):}
          Matriks diagonal dengan entri $\Delta_{v,v} = d(v)$ yang merepresentasikan derajat atau valensi dari verteks $v$.
    \item \textbf{Matriks Laplacian ($\mathbf{Q}$):}
          Didefinisikan melalui hasil kali matriks insidensinya, yaitu $\mathbf{Q} = \mathbf{D}\mathbf{D}^t$. Persamaan ini bernilai ekuivalen dengan
          $ \mathbf{Q} = \Delta - \mathbf{A} $
  \end{itemize}

\end{frame}

\begin{frame}{Tanda pada Polinomial Karakteristik}
  Secara umum, polinomial karakteristik dari suatu matriks $\mathbf{M}$ didefinisikan sebagai $\det(\lambda \mathbf{I} - \mathbf{M})$.

  \vspace{0.5cm}
  Namun, pada Teorema 7.5, polinomial karakteristik dari matriks Laplacian $\mathbf{Q}$ dituliskan dengan tanda positif:
  $$ \tau(\Gamma; \lambda) = \det(\lambda \mathbf{I} + \mathbf{Q}) $$

  \vspace{0.5cm}
  Mengapa menggunakan tanda positif? Alasannya adalah untuk menghindari kemunculan koefisien yang berganti tanda (selang-seling) agar representasi kombinatorialnya menjadi lebih elegan.
\end{frame}

\begin{frame}{Sifat Semidefinit Positif Matriks Laplacian}
  Matriks Laplacian $\mathbf{Q}$ bersifat semidefinit positif, yang berarti seluruh nilai eigennya ($\mu_i$) selalu bernilai non-negatif ($\mu_i \geqslant 0$).

  \begin{itemize}
    \item Jika menggunakan $\det(\lambda \mathbf{I} - \mathbf{Q})$, ekspansinya akan menghasilkan koefisien yang selang-seling: $\lambda^n - q_1 \lambda^{n-1} + q_2 \lambda^{n-2} - \dots$
    \item Jika menggunakan $\det(\lambda \mathbf{I} + \mathbf{Q})$, akar penyelesaiannya bergeser menjadi $-\mu_i$. Persamaannya menjadi $\prod_{i=1}^n (\lambda + \mu_i)$.
  \end{itemize}

  Karena semua $\mu_i$ non-negatif, hasil kalinya menjamin bahwa \textbf{semua koefisien $q_i$ bernilai positif}.
\end{frame}

\begin{frame}{Definisi Fungsi $p(\Phi)$}
  Fungsi $p(\Phi)$ menyatakan total kemungkinan (jumlah cara) untuk memilih \textbf{tepat satu verteks} dari setiap komponen terhubung di dalam hutan $\Phi$.

  \vspace{0.5cm}
  Secara matematis, fungsinya dirumuskan sebagai:
  $$ p(\Phi) = \prod_{j=1}^k |V(C_j)| $$

  di mana $C_j$ adalah komponen-komponen terhubung dalam $\Phi$, dan $|V(C_j)|$ adalah jumlah verteks pada komponen $C_j$.
\end{frame}

\begin{frame}{Teorema Cauchy-Binet}
  \textbf{Pernyataan Teorema:}
  Misalkan $\mathbf{A}$ adalah matriks berukuran $m \times p$ dan $\mathbf{B}$ adalah matriks berukuran $p \times m$ dengan $m \leqslant p$. Maka determinan dari hasil kali $\mathbf{AB}$ diberikan oleh formulasi:
  $$ \det(\mathbf{AB}) = \sum_{S} \det(\mathbf{A}_S) \det(\mathbf{B}^S) $$
  di mana $S$ adalah himpunan bagian dari $\{1, \dots, p\}$ dengan ukuran $m$, $\mathbf{A}_S$ adalah submatriks $m \times m$ yang dibentuk dari kolom-kolom $S$ dari $\mathbf{A}$, dan $\mathbf{B}^S$ adalah submatriks $m \times m$ yang dibentuk dari baris-baris $S$ dari $\mathbf{B}$.
\end{frame}

\begin{frame}{Pembuktian Cauchy-Binet}
  \textbf{Bukti:}
  Bentuk sebuah matriks blok $\mathbf{M}$ yang berukuran $(p+m) \times (p+m)$:
  $$ \mathbf{M} = \begin{pmatrix} \mathbf{I}_p & \mathbf{B} \\ -\mathbf{A} & \mathbf{0}_{m} \end{pmatrix} $$
  Lakukan operasi baris matriks dengan mengalikan matriks identitas termodifikasi sedemikian rupa sehingga:
  $$ \begin{pmatrix} \mathbf{I}_p & \mathbf{0} \\ \mathbf{A} & \mathbf{I}_m \end{pmatrix} \begin{pmatrix} \mathbf{I}_p & \mathbf{B} \\ -\mathbf{A} & \mathbf{0}_m \end{pmatrix} = \begin{pmatrix} \mathbf{I}_p & \mathbf{B} \\ \mathbf{0} & \mathbf{A}\mathbf{B} \end{pmatrix} $$
  Determinan dari ruas kiri adalah $\det(\mathbf{M})$, dan determinan dari ruas kanan adalah $\det(\mathbf{A}\mathbf{B})$. Sehingga jelas bahwa $\det(\mathbf{M}) = \det(\mathbf{A}\mathbf{B})$.

  Lanjutkan dengan mengekspansi Laplace pada matriks $\mathbf{M}$ asli terhadap sejumlah $m$ baris terakhirnya. Minor dari matriks $-\mathbf{A}$ berukuran $m \times m$ akan dikalikan dengan kofaktor dari matriks $\mathbf{B}$. Analisis kombinatorial dari tanda kofaktor ini akan membuktikan bahwa ia menghasilkan tepat jumlahan perkalian determinan submatriks $\sum \det(\mathbf{A}_S) \det(\mathbf{B}^S)$.
  \qed
\end{frame}

\begin{frame}{Teorema 7.5}
  \textbf{Pernyataan Teorema:}
  Koefisien $q_i$ dari polinomial $\tau(\Gamma; \lambda) = \det(\lambda \mathbf{I} + \mathbf{Q})$ diberikan oleh formula:
  $$ q_i = \sum p(\Phi) \quad (1 \leqslant i \leqslant n), $$
  di mana penjumlahan dilakukan atas semua subgraf sisi dari $\Gamma$ yang memiliki $i$ sisi dan merupakan hutan (forest).

  \vspace{0.5cm}
  Secara khusus, berdasarkan observasi dasar polinomial ini, kita memiliki evaluasi:
  $$ q_1 = 2|E\Gamma|, \quad q_{n-1} = n\kappa(\Gamma), \quad q_n = 0 $$
\end{frame}

\begin{frame}{Pembuktian Teorema 7.5 (Bagian 1)}
  \textbf{Bukti:}
  Misalkan $\mathbf{Q}_X$ menotasikan submatriks utama dari $\mathbf{Q}$ di mana baris dan kolomnya bersesuaian dengan verteks dalam subset $X$ dari $V\Gamma$.

  Menurut Proposisi 7.2 dan Proposisi 7.3 diperoleh $q_i = \sum \det \mathbf{Q}_X$, di mana penjumlahan dilakukan atas semua $X$ dengan $|X| = i$.

  Dengan menggunakan identitas relasi matriks $\mathbf{Q} = \mathbf{D}\mathbf{D}^t$, Tinjau bahwa jika $\mathbf{Q}_X$ adalah submatriks $\mathbf{Q}$ terbatas pada baris dan kolom $X$, maka dekomposisinya menjadi:
  $$ \mathbf{Q}_X = \mathbf{D}_X \mathbf{D}_X^t $$
\end{frame}

\begin{frame}{Pembuktian Teorema 7.5 (Bagian 2)}
  Di mana $\mathbf{D}_X$ adalah matriks insidensi $\mathbf{D}$ yang hanya mengambil baris bersesuaian dengan verteks di $X$, namun tetap memuat seluruh kolom (seluruh sisi di $E\Gamma$).

  Misalkan kita memilih himpunan verteks $X$ sedemikian sehingga kardinalitasnya adalah $|X| = i$.

  \begin{itemize}
    \item Matriks $\mathbf{Q}_X$ adalah matriks persegi berukuran $i \times i$.
    \item Matriks $\mathbf{D}_X$ adalah matriks persegi panjang berukuran $i \times |E\Gamma|$.
    \item Matriks $\mathbf{D}_X^t$ adalah matriks persegi panjang berukuran $|E\Gamma| \times i$.
  \end{itemize}

  Kita ingin mencari nilai $\det(\mathbf{Q}_X) = \det(\mathbf{D}_X \mathbf{D}_X^t)$.S
  Karena $\mathbf{D}_X$ bukan matriks persegi, determinannya tidak dapat dieksekusi melalui sifat $\det(\mathbf{A}\mathbf{B}) = \det(\mathbf{A})\det(\mathbf{B})$.

  Namun dengan Teorema Cauchy-Binet, kita bisa memaksa untuk mencari semua kemungkinan submatriks persegi terbesar ($i \times i$) dari matriks persegi panjang $\mathbf{D}_X$.
\end{frame}

\begin{frame}{Pembuktian Teorema 7.5 (Bagian 3)}
  Cara membentuk matriks persegi dari $\mathbf{D}_X$ adalah dengan mempertahankan semua barisnya ($X$), lalu \textbf{memilih tepat $i$ kolom} dari total kolom $|E\Gamma|$. Memilih kolom matriks insidensi sama maknanya dengan memilih \textbf{himpunan subset sisi $Y$} dengan ukuran $|Y| = i$. Submatriks persegi ini dinotasikan sebagai $\mathbf{D}(X, Y)$.

  Menurut Cauchy-Binet, determinan hasil kali adalah jumlahan perkalian dari determinan semua matriks potong persegi tersebut:
  $$ \det(\mathbf{Q}_X) = \sum_{|Y|=i} \det(\mathbf{D}(X, Y)) \cdot \det(\mathbf{D}(X, Y)^t) $$

  Mengingat sifat dasar determinan bahwa nilai determinan suatu matriks sama dengan determinan transposenya ($\det(\mathbf{M}) = \det(\mathbf{M}^t)$), maka:
  $$ \det(\mathbf{Q}_X) = \sum_{|Y|=i} (\det \mathbf{D}(X, Y))^2 $$
\end{frame}

\begin{frame}{Pembuktian Teorema 7.5 (Bagian 4)}
  Penjumlahan ini dilakukan atas semua subset $Y$ dari $E\Gamma$ dengan $|Y| = |X| = i$ (Lema 7.4). Sehingga,
  $$ q_i = \sum_{X, Y} (\det \mathbf{D}(X, Y))^2 $$

  Dari Proposisi 5.3 dijamin nilai determinan $(\det \mathbf{D}(X, Y))^2$ adalah $0$ atau $1$. Lebih lanjut, nilainya adalah $1$ jika dan hanya jika himpunan sisi $Y$ membentuk subgraf hutan $\Phi = \langle Y \rangle$ (Lema 7.4).

  Untuk setiap hutan $\Phi = \langle Y \rangle$, terdapat sebanyak $p(\Phi)$ cara untuk mengabaikan satu verteks dari masing-masing komponen $\Phi$. Akibatnya, ada tepat $p(\Phi)$ buah suku penambah yang bernilai $1$ di dalam ekspresi penjumlahan koefisien $q_i$.
  \qed
\end{frame}

\begin{frame}{Perhitungan Aljabar}
  Matriks Laplacian $\mathbf{Q} = \Delta - \mathbf{A}$ dari $K_4$:
  $$
    \mathbf{Q} = \begin{pmatrix}
      3  & -1 & -1 & -1 \\
      -1 & 3  & -1 & -1 \\
      -1 & -1 & 3  & -1 \\
      -1 & -1 & -1 & 3
    \end{pmatrix}
  $$

  Berdasarkan spektrum graf lengkap, nilai eigen dari Laplacian $K_n$ adalah $0$ (multiplisitas 1) dan $n$ (multiplisitas $n-1$). Untuk $K_4$, nilai eigennya adalah $0, 4, 4, 4$.

  Maka, polinomial karakteristiknya adalah:
  $$ \det(\lambda \mathbf{I} + \mathbf{Q}) = (\lambda + 0)(\lambda + 4)^3 = \lambda(\lambda^3 + 12\lambda^2 + 48\lambda + 64) $$
  $$ = \lambda^4 + 12\lambda^3 + 48\lambda^2 + 64\lambda + 0 $$

  Sehingga: $q_1 = 12$, $q_2 = 48$, $q_3 = 64$, dan $q_4 = 0$.
\end{frame}
