\newcommand{\permcell}[3]{%
  % #1 = x-shift
  % #2 = nilai sigma(i)
  % #3 = indeks i

  % kotak
  \draw[thick, fill=\numcolor{#3}]
    (#1,0) rectangle ++(1,1);

  % lingkaran
  \draw[thick, fill=\numcolor{#2}]
    (#1+0.5,0.5) circle (0.3);

  % angka
  \node at (#1+0.5,0.5) {\small #2};

  % fixed point
  \ifnum#2=#3
    \draw[red!60, very thick] (#1,0) -- (#1+1,1);
    \draw[red!60, very thick] (#1,1) -- (#1+1,0);
  \fi
}
