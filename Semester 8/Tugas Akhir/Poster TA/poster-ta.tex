		\begin{column}{0.39\paperwidth}
			\begin{block}{Abstrak}
				Aljabar max-plus adalah struktur aljabar pada $\Rmax=\mathbb{R}\cup\{\eps\}$ dengan operasi
				$a\oplus b=\max\{a,b\}$ dan $a\otimes b=a+b$. Penelitian ini mengkaji
				komutativitas dua \textit{Latin square} terhadap perkalian max-plus, yaitu
				$A\otimes B=B\otimes A$. Setiap \textit{Latin square} direpresentasikan
				sebagai kombinasi max-plus dari matriks permutasi, sehingga masalah
				komutativitas dapat dianalisis melalui komposisi permutasi penyusunnya.
				Hasil utama penelitian meliputi syarat perlu dari simbol maksimum, syarat
				cukup melalui subgrup abelian dari $\Sn$, kriteria perlu dan cukup melalui
				himpunan superlevel, serta algoritma untuk mengonstruksi pasangan
				\textit{Latin square} komutatif. Algoritma tersebut diterapkan pada ordo
				$n\leq 5$ dan divalidasi kembali secara komputasi dengan menghitung langsung
				hasil perkalian max-plus pada setiap pasangan yang ditemukan.
			\end{block}
		\end{column}

		\begin{column}{0.27\paperwidth}
			\begin{block}{Latar Belakang}
				\begin{itemize}
					\item Aljabar max-plus merupakan salah satu struktur penting dalam aljabar tropikal.
					\item \textit{Latin square} adalah matriks $n\times n$ yang setiap baris dan kolomnya merupakan permutasi dari himpunan simbol yang sama.
					\item Matriks komutatif pada aljabar max-plus berpotensi digunakan dalam kajian aljabar dan kriptografi tropikal.
					\item Penelitian ini berfokus pada syarat komutativitas dan konstruksi pasangan \textit{Latin square} komutatif.
				\end{itemize}

				\vspace{0.4em}
				\textbf{Kontribusi penelitian}
				\begin{itemize}
					\item Menyatakan komutativitas dalam bentuk komposisi permutasi.
					\item Memberikan penyaring awal melalui simbol maksimum dan \textit{centralizer}.
					\item Menyusun algoritma pencarian yang memangkas cabang pencarian sedini mungkin.
				\end{itemize}
			\end{block}
		\end{column}

		\begin{column}{0.25\paperwidth}
			\begin{block}{Tujuan Penelitian}
				\begin{enumerate}
					\item Menentukan syarat perlu komutativitas dua \textit{Latin square}.
					\item Menentukan syarat cukup dan kriteria komutativitas.
					\item Mengonstruksi \textit{Latin square} $B$ sehingga
					      $A\otimes B=B\otimes A$ untuk suatu $A$.
				\end{enumerate}
			\end{block}

			\begin{block}{Batasan Penelitian}
				\begin{itemize}
					\item Simbol yang digunakan adalah $\underline{n}=\{1,2,\ldots,n\}$.
					\item Posisi setiap simbol direpresentasikan sebagai matriks permutasi.
					\item Penerapan algoritma dibatasi pada ordo $n\leq 5$.
				\end{itemize}
			\end{block}
		\end{column}

				\heading{Batasan penelitian}
				Simbol yang digunakan adalah $\underline{n}=\{1,2,\ldots,n\}$ dan
				penerapan algoritma dibatasi pada ordo $n\leq 5$.
			\end{block}

			\begin{block}{Representasi Permutasi}
				Setiap \textit{Latin square} $A\in\LS$ dapat didekomposisi menjadi
				\[
					A=\bigoplus_{i=1}^{n} i\otimes P_{\sigma_i^A}.
				\]
				Jika $B\in\LS$, maka
				\[
					B=\bigoplus_{j=1}^{n} j\otimes P_{\sigma_j^B}.
				\]
				Dengan demikian, hasil perkalian max-plus dianalisis melalui komposisi
				permutasi
				\[
					\sigma_j^B\circ\sigma_i^A
					\quad\text{dan}\quad
					\sigma_i^A\circ\sigma_j^B.
				\]

				\begin{center}
					\begin{tikzpicture}[scale=0.95, every node/.style={font=\small}]
						\drawgridmatrix{Aex}{0}{0}{0.62}{1/1/1,1/2/2,1/3/3,2/1/3,2/2/1,2/3/2,3/1/2,3/2/3,3/3/1}
						\node[above=0.12cm of Aex] {$A$};

						\node[right=0.8cm of Aex] (arrowone) {$\Longrightarrow$};

						\begin{scope}[xshift=4.25cm,yshift=-0.25cm]
							\node[draw=black, fill=cyan!7, rounded corners=2pt, align=center,
								minimum width=4.8cm, minimum height=1.2cm]
							{$\sigma_1^A=(1)$\\
								$\sigma_2^A=(1\;2\;3)$\\
								$\sigma_3^A=(1\;3\;2)$};
						\end{scope}
					\end{tikzpicture}
				\end{center}
			\end{block}

			\begin{block}{Kata Kunci}
				Aljabar Max-Plus, \textit{Latin Square}, Matriks Permutasi, Centralizer,
				Himpunan Superlevel.
			\end{block}

			\begin{thm}{Syarat Perlu}{}
				Jika $A\otimes B=B\otimes A$, maka permutasi yang bersesuaian dengan
				simbol maksimum harus saling komutatif, yaitu
				\[
					\sigma_n^A\circ\sigma_n^B=\sigma_n^B\circ\sigma_n^A.
				\]
				Akibatnya, kandidat awal $\sigma_n^B$ harus dipilih dari
				$C_{S_n}(\sigma_n^A)$.
			\end{thm}

			\begin{thm}{Syarat Cukup}{}
				Jika seluruh permutasi penyusun $A$ dan $B$ berada dalam satu subgrup
				abelian $H\leq S_n$, maka
				\[
					A\otimes B=B\otimes A.
				\]
				Pasangan \textit{Latin square} sirkulan merupakan contoh khusus dari kondisi ini.
			\end{thm}

		\end{column}

		\begin{column}{0.62\paperwidth}

			\begin{block}{Hasil dan Pembahasan}
				\begin{columns}[t]
					\begin{column}{0.32\linewidth}
						\begin{defbox}{Syarat Perlu}
							Jika $A\otimes B=B\otimes A$, maka permutasi simbol maksimum
							harus memenuhi
							\[
								\sigma_n^A\circ\sigma_n^B
								=
								\sigma_n^B\circ\sigma_n^A.
							\]
							Akibatnya, kandidat $\sigma_n^B$ harus berada pada
							$C_{\Sn}(\sigma_n^A)$.
						\end{defbox}
					\end{column}

					\begin{column}{0.32\linewidth}
						\begin{defbox}{Syarat Cukup}
							Jika terdapat subgrup abelian $H\leq \Sn$ sehingga
							\[
								\sigma_i^A\in H
								\quad\text{dan}\quad
								\sigma_j^B\in H
							\]
							untuk setiap $i,j\in\underline{n}$, maka
							\[
								A\otimes B=B\otimes A.
							\]
						\end{defbox}
					\end{column}

					\begin{column}{0.32\linewidth}
						\begin{defbox}{Kriteria Perlu dan Cukup}
							Dua \textit{Latin square} $A,B\in\LS$ komutatif jika dan hanya jika
							\[
								\mathcal{U}_{\ge v}^{AB}
								=
								\mathcal{U}_{\ge v}^{BA}
							\]
							untuk setiap $v\in\{n+2,n+3,\ldots,2n\}$.
						\end{defbox}
					\end{column}
				\end{columns}

				\vspace{0.5em}

				\begin{columns}[t]
					\begin{column}{0.48\linewidth}
						\begin{exampleblock}{Algoritma Pencarian Pasangan}
							Untuk suatu $A\in\LS$, kandidat $B$ dibangun sebagai berikut.
							\begin{enumerate}
								\item Dekomposisikan $A$ ke permutasi penyusun.
								\item Jika permutasi penyusun $A$ saling komutatif, gunakan tahap alternatif.
								\item Jika tidak, pilih kandidat $\sigma_n^B$ dari $C_{\Sn}(\sigma_n^A)$.
								\item Bentuk matriks kandidat parsial $B^*$.
								\item Pilih permutasi berikutnya secara menurun dan periksa superlevel.
								\item Simpan kandidat valid sebagai pasangan komutatif.
							\end{enumerate}

							\[
								B^*=
								\begin{pmatrix}
									1    & 3    & \eps \\
									\eps & 1    & 3    \\
									3    & \eps & 1
								\end{pmatrix}
							\]
							Entri $\eps$ menyatakan posisi yang belum ditentukan.
						\end{exampleblock}
					\end{column}

					\begin{column}{0.48\linewidth}
						\begin{exampleblock}{Contoh Ordo 3}
							\[
								A=
								\begin{pmatrix}
									1 & 2 & 3 \\
									3 & 1 & 2 \\
									2 & 3 & 1
								\end{pmatrix},
								\qquad
								B=
								\begin{pmatrix}
									1 & 3 & 2 \\
									2 & 1 & 3 \\
									3 & 2 & 1
								\end{pmatrix}.
							\]
							Dengan perkalian max-plus diperoleh
							\[
								A\otimes B
								=
								B\otimes A
								=
								\begin{pmatrix}
									6 & 5 & 5 \\
									5 & 6 & 5 \\
									5 & 5 & 6
								\end{pmatrix}.
							\]
							Algoritma juga diterapkan pada ordo $4$ dan $5$.
						\end{exampleblock}
					\end{column}
				\end{columns}
			\end{block}

			\begin{alertblock}{Kesimpulan}
				\begin{enumerate}
					\item Syarat perlu komutativitas diperoleh dari simbol maksimum.
					\item Syarat cukup diberikan oleh subgrup abelian, sedangkan kriteria lengkap diberikan oleh himpunan superlevel.
					\item Algoritma yang disusun dapat digunakan untuk mencari pasangan \textit{Latin square} komutatif untuk $n\leq 5$.
				\end{enumerate}
			\end{alertblock}

			\begin{block}{Validasi Komputasi}
				\begin{center}
					\begin{tabular}{crrl}
						\toprule
						Ordo & Banyak $A$ & Total pasangan $B$ & Status \\
						\midrule
						3    & 12         & 42                 & valid  \\
						4    & 576        & 10080              & valid  \\
						5    & 161280     & 1081560            & valid  \\

				$A\otimes B=B\otimes A$.
			\end{block}
